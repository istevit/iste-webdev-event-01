const siteName = "My Awesome Website";
let clickCount = 0;
const colors = ['#ff6b6b', '#4ecdc4', '#45b7d1', '#96ceb4', '#ffeaa7'];

// Functions
function getRandomColor() {
    return colors[Math.floor(Math.random() * colors.length)];
}

function showMagic() {
    clickCount++;
    
    const output = document.getElementById('output');
    const title = document.getElementById('main-title');
    
    // Change content dynamically
    output.innerHTML = `
        <div class="magic-result fade-in">
            <h3>‚ú® Magic happened! ‚ú®</h3>
            <p>You've clicked ${clickCount} time(s)</p>
            <p>Random color: <span style="color: ${getRandomColor()}">${getRandomColor()}</span></p>
        </div>
    `;
    
    // Change styles
    title.style.color = getRandomColor();
    title.style.transform = 'scale(1.1)';
    
    // Reset transform after animation
    setTimeout(() => {
        title.style.transform = 'scale(1)';
    }, 300);
}

      const viewportSize = document.getElementById('viewportSize');

function updateViewportInfo() {
    const width = window.innerWidth;
    const height = window.innerHeight;
    viewportSize.textContent = `${width} x ${height}px`;
    
    // Detect device type based on width
    if (width < 480) {
        deviceType.textContent = 'Mobile Phone';
    } else if (width < 768) {
        deviceType.textContent = 'Large Phone / Small Tablet';
    } else if (width < 1024) {
        deviceType.textContent = 'Tablet';
    } else {
        deviceType.textContent = 'Desktop / Laptop';
    }
}

// Update on page load and window resize
